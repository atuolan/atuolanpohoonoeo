    <script>
      // 翻页功能JavaScript
      let currentPage = 1;
      const totalPages = 3;
      let isAnimating = false; // 防止动画过程中重复触发

      const prevButton = document.getElementById('prevPage');
      const nextButton = document.getElementById('nextPage');
      const currentPageSpan = document.getElementById('currentPage');
      const totalPagesSpan = document.getElementById('totalPages');
      const navigationButtons = document.querySelector('.navigation-buttons');

      // 更新页面显示
      function updatePageDisplay() {
        // 处理所有页面
        document.querySelectorAll('.page').forEach(page => {
          const pageNum = parseInt(page.getAttribute('data-page'));

          // 移除所有状态类
          page.classList.remove('active', 'turning-left', 'turning-right', 'hidden');

          // 设置适当的类
          if (pageNum === currentPage) {
            page.classList.add('active');
          } else {
            page.classList.add('hidden');
          }
        });

        // 更新页码显示
        currentPageSpan.textContent = currentPage;
        totalPagesSpan.textContent = totalPages;

        // 更新按钮状态 - 第一页只显示右边按钮，最后一页只显示左边按钮
        prevButton.style.display = currentPage === 1 ? 'none' : 'flex';
        nextButton.style.display = currentPage === totalPages ? 'none' : 'flex';

        // 第一頁按鈕放置右側
        if (currentPage === 1) {
          navigationButtons.classList.add('first-page');
        } else {
          navigationButtons.classList.remove('first-page');
        }

        isAnimating = false; // 动画结束
      }

      // 翻页动画
      function turnPage(direction) {
        if (isAnimating) return; // 如果动画正在进行，则忽略新的翻页请求

        const currentPageElement = document.querySelector(`.page-${currentPage}`);

        if (direction === 'next' && currentPage < totalPages) {
          isAnimating = true; // 标记动画开始
          const nextPageElement = document.querySelector(`.page-${currentPage + 1}`);

          // 确保下一页可见但在背面
          nextPageElement.classList.remove('hidden');

          // 添加翻页动画类
          currentPageElement.classList.add('turning-left');

          // 使用transition结束事件而不是setTimeout
          currentPageElement.addEventListener(
            'transitionend',
            function handler() {
              currentPageElement.removeEventListener('transitionend', handler);
              currentPage++;
              updatePageDisplay();
            },
            { once: true },
          ); // once:true确保事件只触发一次
        } else if (direction === 'prev' && currentPage > 1) {
          isAnimating = true; // 标记动画开始
          const prevPageElement = document.querySelector(`.page-${currentPage - 1}`);

          // 确保上一页可见
          prevPageElement.classList.remove('hidden');

          // 添加翻页动画类
          currentPageElement.classList.add('turning-right');

          // 使用transition结束事件
          currentPageElement.addEventListener(
            'transitionend',
            function handler() {
              currentPageElement.removeEventListener('transitionend', handler);
              currentPage--;
              updatePageDisplay();
            },
            { once: true },
          );
        }
      }

      // 事件监听器
      nextButton.addEventListener('click', () => turnPage('next'));
      prevButton.addEventListener('click', () => turnPage('prev'));

      // 键盘支持
      document.addEventListener('keydown', e => {
        if (e.key === 'ArrowLeft') {
          turnPage('prev');
        } else if (e.key === 'ArrowRight') {
          turnPage('next');
        }
      });

      // 初始化 - 将所有非当前页面隐藏
      document.addEventListener('DOMContentLoaded', function () {
        document.querySelectorAll('.page').forEach(page => {
          const pageNum = parseInt(page.getAttribute('data-page'));
          if (pageNum !== currentPage) {
            page.classList.add('hidden');
          }
        });
        updatePageDisplay();
      });

      // 如果DOM已加载，立即初始化
      if (document.readyState !== 'loading') {
        document.querySelectorAll('.page').forEach(page => {
          const pageNum = parseInt(page.getAttribute('data-page'));
          if (pageNum !== currentPage) {
            page.classList.add('hidden');
          }
        });
        updatePageDisplay();
      }
    </script>
