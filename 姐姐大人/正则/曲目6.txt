<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>音乐播放器</title>
    <style>
        /* ========== 頁面整體佈局樣式 ========== */
        body {
            display: flex;              /* 使用彈性佈局 */
            justify-content: center;    /* 水平置中 */
            align-items: center;        /* 垂直置中 */
            margin: 0;                  /* 移除預設邊距 */
            padding: 20px;              /* 頁面內邊距 */
        }

        /* ========== 音樂播放器主容器樣式 ========== */
        .music-player {
            width: 100%;                /* 寬度100% */
            max-width: 220px;           /* 最大寬度220px */
            height: 50px;               /* 固定高度50px */
            background: linear-gradient(to right, #bcd4e4ff, #6a85a0ff);  /* 從左到右的漸層背景 */
            border-radius: 6px;         /* 圓角6px */
            padding: 8px 12px;          /* 內邊距：上下8px，左右12px */
            display: flex;              /* 使用彈性佈局 */
            align-items: center;        /* 垂直置中對齊 */
            gap: 12px;                  /* 子元素間距12px */
            color: white;               /* 文字顏色白色 */
            font-family: "DFKai-SB", "標楷體", serif;  /* 字體設定：標楷體 */
        }

        /* ========== 播放按鈕樣式 ========== */
        .play-btn {
            background: none;           /* 無背景 */
            border: none;               /* 無邊框 */
            color: white;               /* 文字顏色白色 */
            font-size: 22px;            /* 字體大小22px */
            cursor: pointer;            /* 滑鼠指標變手型 */
            padding: 5px;               /* 內邊距5px */
            min-width: 36px;            /* 最小寬度36px */
            flex-shrink: 0;             /* 不允許收縮 */
            margin-left: 10px;          /* 左邊距10px（向右移動） */
        }

        /* ========== 播放按鈕滑鼠懸停效果 ========== */
        .play-btn:hover {
            opacity: 0.8;               /* 透明度變為80% */
        }

        /* ========== 主要內容區域樣式 ========== */
        .main-content {
            flex: 1;                    /* 佔據剩餘空間 */
            display: flex;              /* 使用彈性佈局 */
            justify-content: center;    /* 水平置中 */
            align-items: center;        /* 垂直置中 */
        }

        /* ========== 歌曲標題樣式 ========== */
        .song-title {
            font-size: 16px;            /* 字體大小16px */
            font-weight: bold;          /* 粗體 */
            margin: 0;                  /* 移除邊距 */
            text-align: center;         /* 文字置中 */
            white-space: nowrap;        /* 不換行 */
            overflow: hidden;           /* 隱藏溢出內容 */
            text-overflow: ellipsis;    /* 溢出時顯示省略號 */
        }
    </style>
</head>
<body>
    <!-- ========== HTML 結構區域 ========== -->
    <!-- 音樂播放器主容器 -->
    <div class="music-player">
        <!-- 播放/暫停按鈕 -->
        <button class="play-btn" id="playBtn">▶</button>

        <!-- 主要內容區域 -->
        <div class="main-content">
            <!-- 歌曲標題顯示 -->
            <div class="song-title">別再鬧了</div>
        </div>
    </div>

    <!-- ========== JavaScript 功能區域 ========== -->
    <script>
        $(function() {
            // ========== DOM 元素選取 ==========
            const playBtn = $('.play-btn');        // 播放按鈕
            const songTitle = $('.song-title');    // 歌曲標題

            // ========== 音樂設定 ==========
            const songUrl = 'https://aguacloudreve.aguacloud.uk/f/rDFp/%5B%E9%8B%BC%E7%90%B4%E9%9F%B3%E6%A8%82%5D%20%E6%AF%9B%E4%B8%8D%E6%98%93-%E5%88%A5%E5%86%8D%E9%AC%A7%E4%BA%86.mp3';  // 音樂檔案URL
            const songName = '別再鬧了';                // 歌曲名稱

            // ========== 狀態變數 ==========
            let isPlaying = false;              // 播放狀態標記

            // ========== TavernHelper API 檢查 ==========
            if (typeof TavernHelper === 'undefined') {
                songTitle.text('TavernHelper 不可用');
                return;
            }

            // ========== 播放/暫停按鈕事件處理 ==========
            playBtn.on('click', async function() {
                try {
                    if (isPlaying) {
                        // ========== 暫停音樂 ==========
                        pauseAudio('bgm');
                        playBtn.text('▶');         // 改變按鈕為播放圖示
                        isPlaying = false;         // 更新播放狀態
                    } else {
                        // ========== 播放音樂 ==========
                        playAudio('bgm', { title: songName, url: songUrl });
                        playBtn.text('⏸');        // 改變按鈕為暫停圖示
                        isPlaying = true;         // 更新播放狀態
                    }
                } catch (error) {
                    console.error('播放控制失敗:', error);
                }
            });

            // ========== 自動播放功能 ==========
            async function autoPlay() {
                try {
                    // 設置單曲循環模式
                    setAudioSettings('bgm', { mode: 'repeat_one' });

                    // 使用酒馆助手 API 播放
                    playAudio('bgm', { title: songName, url: songUrl });

                    // 更新狀態和按鈕
                    playBtn.text('⏸');
                    isPlaying = true;

                    console.log('自動播放成功');
                } catch (error) {
                    console.log('自動播放失敗:', error);
                }
            }

            // 執行自動播放
            autoPlay();
        });
    </script>
</body>
</html>
